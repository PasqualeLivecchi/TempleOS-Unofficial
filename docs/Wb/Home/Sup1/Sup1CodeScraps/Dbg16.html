DBG16_PUT_CHAR::
        MOV     AH,0xE
        MOV     BL,7 //Might be foreground color on some BIOS's
        MOV     BH,U8 [DBG16_PAGE]
        INT     0x10
DBG16_RET:
        RET

DBG16_PUTS::
@@1:    LODSB
        TEST    AL,AL
        JZ      DBG16_RET
        CALL    DBG16_PUT_CHAR
        JMP     @@1

DBG16_PAGE:
        DU8     0;
DBG16_INIT::
        MOV     AH,0xF
        INT     0x10
        MOV     U8 [DBG16_PAGE],BH //Video page
        RET

DBG16_NUM:
        DU8     '0';
DBG16_PT::
        MOV     AL,U8 [DBG16_NUM]
        INC     U8 [DBG16_NUM]
        CALL    DBG16_PUT_CHAR
        XOR     EAX,EAX
@@1:    DEC     EAX
        JNZ     @@1
        RET
