//5 has graphics

F64 t0=Beat;

U0 DrawIt(CTask *task,CDC *dc)
{
  I64 i,j,w=task->pix_width,h=task->pix_height;
  F64 dt=Beat-t0,num;

  num=MinI64(Exp(0.9*dt),3000);
  dc->color=LTGRAY;
  for (i=0;i<num;i++) {
    j=RandU32;
    GrPlot(dc,j.u16[0]%w,j.u16[1]%h);
  }

  num=MinI64(Exp(0.6*dt),6000);
  dc->color=DKGRAY;
  for (i=0;i<num;i++) {
    j=RandU32;
    GrPlot(dc,j.u16[0]%w,j.u16[1]%h);
  }

  num=MinI64(Exp(0.3*dt),9000);
  dc->color=BLACK;
  for (i=0;i<num;i++) {
    j=RandU32;
    GrPlot(dc,j.u16[0]%w,j.u16[1]%h);
  }
}
