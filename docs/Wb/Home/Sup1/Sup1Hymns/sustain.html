F64 t0=Beat;

U0 DrawIt(CTask *,CDC *dc)
{
  F64 tt=Wrap(pi/2*(Beat-t0),0);
  dc->flags|=DCF_TRANSFORMATION;
  Mat4x4RotX(dc->r,0.3*Sin(tt/4));
  if (pi/2<=tt<=3*pi/2)
    tt=pi-tt;
  Mat4x4RotY(dc->r,tt);
  dc->x=325;
  dc->y=35;
  Sprite3(dc,0,0,0,<1>);
}
