U0 DrawIt(CTask *task,CDC *dc)
{
  I64 i,w=task->pix_width,h=task->pix_height;
  dc->color=BROWN;
  for (i=60*Saw(Beat,20);i<w;i+=20)
    GrLine(dc,i,h,w,h-i/4);
  dc->color=BLACK;
  GrLine(dc,0,h-1,w,h-1);
  GrLine(dc,w-1,0,w-1,h);
}
