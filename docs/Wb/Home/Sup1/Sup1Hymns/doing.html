F64 t0=Beat;
U0 DrawIt(CTask *,CDC *dc)
{
  F64 dt=Beat-t0;
  dc->flags|=DCF_TRANSFORMATION;
  DCDepthBufAlloc(dc);
  dc->x=320;
  dc->y=27;
  dc->z=GR_Z_ALL;
  Mat4x4RotX(dc->r,0.140*pi);
  Mat4x4RotZ(dc->r,0.250*pi*Sin(pi*dt));
  Mat4x4RotY(dc->r,0.230*pi);
  if (dt>16)
    Mat4x4RotY(dc->r,0.5*Sin(pi/2*dt));
  Sprite3(dc,0,0,0,<1>);
}
