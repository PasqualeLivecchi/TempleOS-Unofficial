F64 t0=Beat;
U0 DrawIt(CTask *,CDC *dc)
{
  F64 dt=Beat-t0;
  U8 *tmps;
  DCDepthBufAlloc(dc);
  dc->x=280;
  dc->y=30;
  dc->z=GR_Z_ALL;
  dc->flags|=DCF_TRANSFORMATION;
  if (dt<16) {
    Mat4x4RotZ(dc->r,(dt+5)*pi/2);
    Mat4x4RotY(dc->r,FullTri(dt,8)*pi/8);
  } else
    Mat4x4RotZ(dc->r,5*pi/2);
  Mat4x4RotX(dc->r,-11*pi/8);
  Mat4x4Scale(dc->r,0.75);
  if (dt<16)
    Sprite3(dc,0,0,0,<1>);
  else if (16<dt<17 || 23<dt<25 || 31<dt) {
    tmps=SpriteInterpolate(Tri(dt-16,2.0),<1>,<2>);
    Sprite3(dc,0,0,0,tmps);
    Free(tmps);
  } else
    Sprite3(dc,0,0,0,<2>);
}
