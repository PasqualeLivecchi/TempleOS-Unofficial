F64 t0=Beat;

#define WOOD_X  220
#define WOOD_Y  45

U0 DrawIt(CTask *,CDC *dc)
{
  U8 *img;
  F64 dt=Beat-t0;
  I64 r;

  if (dt<8)
    r=3;
  else if (dt<18)
    r=dt-5;
  else
    r=18-5;
  dc->color=BLUE;
  GrEllipse  (dc,WOOD_X,WOOD_Y,r,1.5*r,pi/8);
  GrFloodFill(dc,WOOD_X,WOOD_Y);

  if (dt<20)
    img=<1>;
  else if (dt%0.4>.2)
    img=<2>;
  else
    img=<3>;
  Sprite3(dc,WOOD_X,WOOD_Y,0,img);
}
