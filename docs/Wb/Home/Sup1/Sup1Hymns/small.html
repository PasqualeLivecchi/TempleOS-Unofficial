#define FRAMES 8

U8 *imgs[FRAMES]={
  <1>,<2>,<3>,<4>,
  <5>,<4>,<3>,<1>
};

F64 t0=Beat;

U0 DrawIt(CTask *task,CDC *dc)
{
  U8 *tmps;
  F64 tt=Beat,dt=Beat-t0;
  tmps=SpriteInterpolate(tt%1.0,imgs[tt%FRAMES],imgs[(tt+1)%FRAMES]);
  Sprite3(dc,180,40,0,tmps);
  Free(tmps);
  if (dt>24.0)
    Sprite3(dc,task->pix_width-12.0*(dt-24.0),40,0,<6>);
}
