F64 t0=Beat;
U0 DrawIt(CTask *,CDC *dc)
{
  F64 dt=Beat-t0,a_x,a_y;
  if (dt<8) {
    a_y=pi/2;
    a_x=Wrap(pi*dt/4,0);
  } else if (dt<32) {
    a_x=0;
    a_y=Wrap(pi*dt/16,0);
    if (pi/2<=a_y<=3*pi/2) a_y=-a_y;
  } else if (dt<44) {
    a_y=0;
    a_x=0;
  } else {
    a_x=0;
    a_y=Wrap(pi*dt/8+pi/2,0);
    if (pi/2<=a_y<=3*pi/2) a_y=-a_y;
  }
  dc->flags|=DCF_TRANSFORMATION;
  Mat4x4RotX(dc->r,a_x);
  Mat4x4RotY(dc->r,a_y);
  dc->x=325;
  dc->y=35;
  Sprite3(dc,0,0,0,<1>);
}
